 (defun ari/exwm-float-hm-hero ()
  (when (and exwm-class-name
             (string= exwm-class-name "hm_hero"))
    (exwm-floating-toggle-floating)
    (exwm-layout-hide-mode-line)))

 (add-hook 'exwm-manage-finish-hook #'ari/exwm-float-hm-hero)
